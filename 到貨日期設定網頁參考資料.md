    <script>
        // 載入當前設定
        async function loadCurrentSettings() {
            try {
                const response = await fetch('./api/delivery_settings.php');
                const result = await response.json();

                if (result.success) {
                    const settings = result.data;

                    // 更新表單
                    document.getElementById('startDate').value = settings.delivery_period.start_date;
                    document.getElementById('endDate').value = settings.delivery_period.end_date;
                    document.getElementById('description').value = settings.delivery_period.description || '';

                    // 更新當前設定顯示
                    updateSettingsDisplay(settings);

                    // 更新預覽
                    updatePreview();
                } else {
                    showStatus('載入設定失敗: ' + result.message, 'error');
                }
            } catch (error) {
                showStatus('載入設定錯誤: ' + error.message, 'error');
            }
        }

        // 格式化更新時間（台北時區）
        function formatUpdateTime(timeString) {
            try {
                // 假設 timeString 已經是台北時區的時間
                const date = new Date(timeString);

                // 格式化為台北時區顯示
                const options = {
                    year: 'numeric',
                    month: '2-digit',
                    day: '2-digit',
                    hour: '2-digit',
                    minute: '2-digit',
                    second: '2-digit',
                    timeZone: 'Asia/Taipei',
                    hour12: false
                };

                const formatter = new Intl.DateTimeFormat('zh-TW', options);
                const formattedDate = formatter.format(date);

                return `${formattedDate}`;
            } catch (error) {
                console.error('時間格式化錯誤:', error);
                return timeString;
            }
        }

        // 更新設定顯示
        function updateSettingsDisplay(settings) {
            const period = settings.delivery_period;
            const formattedTime = formatUpdateTime(period.updated_at);

            const html = `
                <div class="setting-item">
                    <span class="setting-label">開始日期：</span>
                    <span class="setting-value">${period.start_date}</span>
                </div>
                <div class="setting-item">
                    <span class="setting-label">結束日期：</span>
                    <span class="setting-value">${period.end_date}</span>
                </div>
                <div class="setting-item">
                    <span class="setting-label">設定說明：</span>
                    <span class="setting-value">${period.description || '無'}</span>
                </div>
                <div class="setting-item">
                    <span class="setting-label">最後更新：</span>
                    <span class="setting-value">${formattedTime} (${period.updated_by})</span>
                </div>
            `;
            document.getElementById('settingsDisplay').innerHTML = html;
        }

        // 更新預覽
        async function updatePreview() {
            try {
                // 載入宅配到府日期
                const homeResponse = await fetch('./api/delivery_settings.php?action=available_dates&delivery_method=宅配到府');
                const homeResult = await homeResponse.json();

                if (homeResult.success) {
                    updateDateList('homeDeliveryDates', homeResult.data);
                }

                // 載入7-11門市日期
                const storeResponse = await fetch('./api/delivery_settings.php?action=available_dates&delivery_method=超商取貨');
                const storeResult = await storeResponse.json();

                if (storeResult.success) {
                    updateDateList('storePickupDates', storeResult.data);
                }
            } catch (error) {
                console.error('更新預覽失敗:', error);
            }
        }

        // 更新日期列表
        function updateDateList(elementId, dates) {
            const container = document.getElementById(elementId);
            const html = dates.map(date => {
                const isSunday = date.dayOfWeek === 0;
                const className = isSunday ? 'date-item sunday' : 'date-item';
                return `<div class="${className}">${date.display}</div>`;
            }).join('');
            container.innerHTML = html;
        }

        // 表單提交
        document.getElementById('settingsForm').addEventListener('submit', async (e) => {
            e.preventDefault();

            const formData = new FormData(e.target);
            const data = Object.fromEntries(formData.entries());
            data.updated_by = 'admin';

            try {
                const response = await fetch('./api/delivery_settings.php', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify(data)
                });

                const result = await response.json();

                if (result.success) {
                    showStatus('設定更新成功！', 'success');
                    updateSettingsDisplay(result.data);
                    updatePreview();
                } else {
                    showStatus('更新失敗: ' + result.message, 'error');
                }
            } catch (error) {
                showStatus('更新錯誤: ' + error.message, 'error');
            }
        });

        // 顯示狀態訊息
        function showStatus(message, type) {
            const statusDiv = document.getElementById('statusMessage');
            statusDiv.className = `status ${type}`;
            statusDiv.textContent = message;

            // 3秒後自動隱藏
            setTimeout(() => {
                statusDiv.textContent = '';
                statusDiv.className = '';
            }, 3000);
        }

        // 頁面載入時初始化
        document.addEventListener('DOMContentLoaded', () => {
            loadCurrentSettings();
        });
    </script>
